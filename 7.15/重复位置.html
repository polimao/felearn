字符串重复位置

-----题
查找字符串“后盾视频下载地址:bbs.houdunwang.com，论坛地址：bbs.houdunwang.com,官方网址：www.houdunwang.com”
将所有”houdunwang”出现的位置，打印出来

1、演示

2、代码

<script type="text/javascript"> var str1 = '后盾视频下载地址:bbs.houdunwang.com，论坛地址：bbs.houdunwang.com,官方网址：www.houdunwang.com'; var str2 = 'houdunwang'; function print(str1, str2) { var re = []; var i = 0 while (str1.match(str2)) { var adr = str1.indexOf(str2); re.push(adr); str1 = str1.replace(str1[adr], '$'); i = i + 1 document.write('第' + i + '次适配得到位置：' + re + '<br>' + str1 + '<br><br>'); } return re; } var result = print(str1, str2) document.write('最终输出结果：' + result) </script>
