数组中的元素排序

方法一、最大 清除为0 最大……

1、演示

2、代码

<script type="text/javascript"> var arr = [35, 12, 12, 345, 7, 87, 2, 3, 88, 2344]; document.write(arr + '<br>'); var re = []; var max = 0; for (var i = 0; i < arr.length; i++) { for (var k = 0; k < arr.length; k++) { if (arr[k] > max) { max = arr[k]; b = k; }; }; re[re.length] = max; arr[b] = 0; max = 0; document.write(i + '次re=[' + re + ']<br>') document.write('arr=[' + arr + ']<br><br>') }; document.write('<h3>最终re=[' + re + ']</h1>') </script>

方法二、大数小数交换位置

1、演示

2、代码

<script type="text/javascript"> var arr = [35, 12, 12, 345, 7, 87, 2, 3, 88, 2344] document.write(arr + '<br />') var i = 0, len = arr.length, j, d; for (i; i < len; i++) { for (j = 0; j < len; j++) { if (arr[i] > arr[j]) { d = arr[j]; arr[j] = arr[i]; arr[i] = d; } } document.write('第' + i + '次：arr=[' + arr + ']<br><br>') } document.write('<h3>最终re=[' + arr + ']</h1>') </script>
